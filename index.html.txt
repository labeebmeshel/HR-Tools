<script>
    // التنقل بين التبويبات
    function openTab(evt, tabName) {
        let i, tabcontent, tablinks;
        tabcontent = document.getElementsByClassName("tab-content");
        for (i = 0; i < tabcontent.length; i++) { tabcontent[i].style.display = "none"; tabcontent[i].classList.remove("active"); }
        tablinks = document.getElementsByClassName("tab-btn");
        for (i = 0; i < tablinks.length; i++) { tablinks[i].classList.remove("active"); }
        document.getElementById(tabName).style.display = "block";
        document.getElementById(tabName).classList.add("active");
        evt.currentTarget.classList.add("active");
    }

    // منطق الحسابات
    function tax(income) {
        if(income <= 40000) return 0;
        if(income <= 55000) return (income - 40000) * 0.10;
        if(income <= 70000) return (income - 55000) * 0.15 + 1500;
        if(income <= 200000) return (income - 70000) * 0.20 + 3750;
        if(income <= 400000) return (income - 200000) * 0.225 + 29750;
        return (income - 400000) * 0.25 + 74750;
    }

    function calculate() {
        const total = Number(document.getElementById('salary').value || 0);
        const realAllow = Number(document.getElementById('allowances').value || 0);
        
        let theoreticalInsBase = total / 1.30;
        let maxLegalAllow = total - theoreticalInsBase;
        
        let finalAllow = realAllow > maxLegalAllow ? maxLegalAllow : realAllow;
        let finalInsBase = total - finalAllow;

        if(finalInsBase < 2700) finalInsBase = 2700;
        if(finalInsBase > 16700) finalInsBase = 16700;

        const insVal = finalInsBase * 0.11;
        const taxableIncome = Math.max(0, (total - insVal) * 12 - 20000);
        const monthlyTax = tax(taxableIncome) / 12;
        const net = total - insVal - monthlyTax;

        document.getElementById('resGross').textContent = total.toFixed(2);
        document.getElementById('resAllow').textContent = finalAllow.toFixed(2);
        document.getElementById('resInsBase').textContent = finalInsBase.toFixed(2);
        document.getElementById('resInsVal').textContent = insVal.toFixed(2);
        document.getElementById('resTax').textContent = monthlyTax.toFixed(2);
        document.getElementById('resNet').textContent = net.toFixed(2);
        document.getElementById('msg').textContent = realAllow > maxLegalAllow ? "ملاحظة: تم تطبيق حد الإعفاء القانوني للبدلات (30%)" : "تم الحساب بناءً على البدلات الحقيقية";
    }
</script>
