function calculateInsurance(basicSalary, allowances) { // 1. حساب الأجر الشامل الأساسي let grossSalary = basicSalary + allowances; // الحد الأقصى القانوني للأجر التأميني const MAX_INSURANCE_WAGE = 16700; // 2. الحسبة المبدئية للأجر الخاضع للتأمين بدون البدلات المعفاة // القانون يعفي البدلات بحد أقصى 30% من الأجر التأميني المستهدف // برمجياً: الأجر التأميني المبدئي = الأجر الأساسي + البدلات غير المعفاة let simulatedInsuranceWage = basicSalary; let maxAllowableExemption = 0; // احتساب سقف الإعفاء بناءً على الراتب الأساسي والحد الأقصى if (basicSalary >= MAX_INSURANCE_WAGE) { simulatedInsuranceWage = MAX_INSURANCE_WAGE; } else { // إذا كان الأساسي أقل، نختبر قدرة البدلات على ملء الوعاء التأميني // المعادلة الرياضية العكسية للإعفاء (البدلات المعفاة = الشامل - التأميني) // بحد أقصى 30% من الأجر التأميني نفسه، مما يعني أن الأساسي يمثل 70% على الأقل إذا كانت البدلات ضخمة let potentialInsurance = basicSalary / 0.7; if (potentialInsurance > MAX_INSURANCE_WAGE) { simulatedInsuranceWage = MAX_INSURANCE_WAGE; } else { // إذا كان الإجمالي ككل صغير، نخصم البدلات المعفاة الحقيقية maxAllowableExemption = potentialInsurance * 0.3; if (allowances <= maxAllowableExemption) { simulatedInsuranceWage = basicSalary; // البدلات معفاة بالكامل } else { simulatedInsuranceWage = grossSalary - maxAllowableExemption; } } } // تطبيق الحد الأقصى الصارم (16,700) let finalInsuranceWage = Math.min(simulatedInsuranceWage, MAX_INSURANCE_WAGE); // تقريب الأرقام لأقرب جنيه مثل جداول التأمينات الرسمية finalInsuranceWage = Math.round(finalInsuranceWage); // 3. الأجر الشامل طبقاً لـ س1 (وهو الأجر التأميني النهائي + أي مبالغ أخرى غير خاضعة لكن تظهر في الاستمارة) let finalGrossS1 = grossSalary; return { grossSalary: grossSalary, // الأجر الشامل الحقيقي insuranceWage: finalInsuranceWage, // الأجر التأميني (بحد أقصى 16700) grossS1: finalGrossS1 // الأجر الشامل طبقاً لـ س1 }; }
