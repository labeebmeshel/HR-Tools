function calcInsurance(base, allow, insMode, manualWage) { if (insMode === 'none') return { finalInsBase: 0, finalExemptAllow: 0 }; if (insMode === 'manual') { let f = Math.max(2700, Math.min(16700, manualWage)); let e = Math.max(0, Math.min(allow, (base + allow) - f)); return { finalInsBase: Math.round(f), finalExemptAllow: Math.round(e) }; } const MAX = 16700; const MIN = 2700; let totalGross = base + allow; // الأجر الشامل (أساسي + بدلات) let f = base; // الشرط الصريح: لو البدلات المدخلة أكبر من 30% من الراتب الأساسي (الأجر التأميني المفترض) if (allow > (base * 0.3)) { // نقسم الأجر الشامل على 1.30 يطلع الأجر التأميني المقدر f = totalGross / 1.30; } else { // لو البدلات في الحدود الآمنة، الأجر التأميني هو الأساسي بالكامل والبدلات تعفى بالكامل f = base; } // تطبيق قاعدة الحد الأقصى 16700 والحد الأدنى 2700 f = Math.max(MIN, Math.min(MAX, f)); // البدلات المعفاة تأمينياً الفعالية هي الإجمالي ناقص الأجر التأميني الثابت let e = Math.max(0, Math.min(allow, totalGross - f)); return { finalInsBase: Math.round(f), finalExemptAllow: Math.round(e) }; }
