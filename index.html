async function updateUI() { try { // قراءة الملف من الجيت هب (استخدم رابط raw للملف) const response = await fetch('config.json?v=' + new Date().getTime()); const config = await response.json(); // 1. التحكم في البطاقات (Cards) if (!config.cv_visible) document.getElementById('tool-cv').style.display = 'none'; if (!config.tax_visible) document.getElementById('tool-tax').style.display = 'none'; if (!config.s1_visible) document.getElementById('tool-s1').style.display = 'none'; if (!config.tafqit_visible) document.getElementById('tool-tafqit').style.display = 'none'; if (!config.performance_visible) document.getElementById('tool-feedback').style.display = 'none'; // 2. إخفاء التابات العلوية (Navbar) // سنبحث عن التابة بناءً على النص أو الـ ID const tabs = window.parent.document.querySelectorAll('.nav-tab'); // افترضنا أن هذا كلاس التابة tabs.forEach(tab => { if (tab.innerText.includes("السيرة") && !config.cv_visible) tab.style.display = 'none'; if (tab.innerText.includes("الضرائب") && !config.tax_visible) tab.style.display = 'none'; // ... وهكذا لبقية التابات }); } catch (error) { console.log("Error loading config:", error); } } // تشغيل الدالة بمجرد فتح الصفحة window.onload = updateUI;
